// SELECCIÓN ALEATORIA DE 10 PREGUNTAS Y MEZCLA DE OPCIONES
// ============================================================
var preguntasSeleccionadas = mezclar(bankPreguntas).slice(0, 10);
var preguntasRenderizadas = [];

for (var i = 0; i < preguntasSeleccionadas.length; i++) {
    var p = preguntasSeleccionadas[i];
    var opcionesMezcladas = mezclar(p.opciones);
    var letraCorrecta = String.fromCharCode(65 + opcionesMezcladas.indexOf(p.respuestaCorrecta));
    preguntasRenderizadas.push({
        id: p.id,
        pregunta: p.pregunta,
        opciones: opcionesMezcladas,
        letraCorrecta: letraCorrecta
    });
}

// ============================================================
// RENDERIZAR PREGUNTAS EN EL DOM
// ============================================================
var qContainer = document.getElementById('randomQuestions');
for (var i = 0; i < preguntasRenderizadas.length; i++) {
    var pr = preguntasRenderizadas[i];
    var div = document.createElement('div');
    div.className = 'question';
    var html = '<div class="question-text">' + (i + 5) + '. ' + pr.pregunta + ' <span class="required">*</span></div><div class="options">';
    for (var j = 0; j < pr.opciones.length; j++) {
        var letra = String.fromCharCode(65 + j);
        html += '<label class="option"><input type="radio" name="q' + pr.id + '" value="' + letra + '" required><span>' + letra + '. ' + pr.opciones[j] + '</span></label>';
    }
    html += '</div>';
    div.innerHTML = html;
    qContainer.appendChild(div);
}

// ============================================================
// BARRA DE PROGRESO
// ============================================================
var form = document.getElementById('evaluationForm');
var progressBar = document.getElementById('progressBar');
form.addEventListener('change', function() {
    var filled = Array.from(form.elements).filter(function(el) {
        return (el.type === 'text' && el.value.trim() !== '') ||
               (el.type === 'email' && el.value.trim() !== '') ||
               (el.type === 'radio' && el.checked);
    }).length;
    progressBar.style.width = (filled / 14) * 100 + '%';
});

// ============================================================
// ENVÍO DEL FORMULARIO
// ============================================================
form.addEventListener('submit', function(e) {
    e.preventDefault();

    var submitBtn = document.getElementById('submitBtn');
    var loading = document.getElementById('loading');
    var resultContainer = document.getElementById('resultContainer');

    submitBtn.disabled = true;
    submitBtn.textContent = 'Enviando...';
    loading.style.display = 'block';

    var correctas = 0;
    var respuestasDetalle = [];

    for (var i = 0; i < preguntasRenderizadas.length; i++) {
        var pr = preguntasRenderizadas[i];
        var checked = document.querySelector('input[name="q' + pr.id + '"]:checked');
        var respuesta = checked ? checked.value : '';
        var esCorrecta = (respuesta === pr.letraCorrecta);
        if (esCorrecta) correctas++;
        respuestasDetalle.push({
            numero: i + 1,
            pregunta: pr.pregunta,
            respuesta_dada: respuesta,
            respuesta_correcta: pr.letraCorrecta,
            es_correcta: esCorrecta
        });
    }

    var porcentaje = (correctas / 10) * 100;
    var aprobado = porcentaje >= 80;

    var formData = {
        timestamp: new Date().toLocaleString('es-CR', {timeZone: 'America/Costa_Rica'}),
        nombre: document.getElementById('nombre').value,
        cedula: document.getElementById('cedula').value,
        correo: document.getElementById('correo').value,
        empresa: document.getElementById('empresa').value,
        puntaje: correctas + '/10',
        porcentaje: porcentaje.toFixed(1) + '%',
        calificacion: aprobado ? 'APROBADO' : 'REPROBADO',
        preguntas_ids: preguntasRenderizadas.map(function(pr) { return pr.id; }).join(', '),
        respuestas: respuestasDetalle
    };

    fetch(GOOGLE_SCRIPT_URL, {
        method: 'POST',
        mode: 'no-cors',
        headers: {'Content-Type': 'application/json'},
        body: JSON.stringify(formData)
    }).then(function() {
        loading.style.display = 'none';
        form.style.display = 'none';
        resultContainer.style.display = 'block';

        document.getElementById('scoreDisplay').textContent = correctas + '/10';
        document.getElementById('scoreMessage').textContent = porcentaje.toFixed(1) + '%';
        document.getElementById('scoreBadge').textContent = aprobado ? 'APROBADO ✓' : 'REPROBADO ✗';
        document.getElementById('scoreBadge').className = 'success-badge ' + (aprobado ? 'badge-aprobado' : 'badge-reprobado');

        var detallesHTML = '<h3 style="margin-bottom:15px;">📊 Detalles de la Evaluación</h3>';
        detallesHTML += '<p><strong>Nombre:</strong> ' + formData.nombre + '</p>';
        detallesHTML += '<p><strong>Cédula:</strong> ' + formData.cedula + '</p>';
        detallesHTML += '<p><strong>Empresa:</strong> ' + formData.empresa + '</p>';
        detallesHTML += '<p style="margin-top:10px;"><strong>Respuestas correctas:</strong> ' + correctas + ' de 10</p>';
        detallesHTML += '<p><strong>Nota mínima requerida:</strong> 80%</p>';
        detallesHTML += '<hr style="margin:15px 0;">';
        detallesHTML += '<p style="color:' + (aprobado ? '#2ecc71' : '#e74c3c') + '; font-weight:bold;">' +
            (aprobado ? '¡Felicitaciones! Has aprobado la evaluación de seguridad.' :
             'No alcanzaste la nota mínima del 80%. Debes repasar los conceptos y volver a realizar la evaluación.') + '</p>';
        document.getElementById('scoreDetails').innerHTML = detallesHTML;

    }).catch(function(error) {
        console.error('Error:', error);
        loading.style.display = 'none';
        alert('Error al enviar. Por favor intenta nuevamente.');
        submitBtn.disabled = false;
        submitBtn.textContent = '✓ Enviar Evaluación';
    });
});
